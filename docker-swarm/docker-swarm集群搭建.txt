四、编写yml文件，部署服务
	1.编写 base-stack.yml
		version: "3"
		services:

		  redis:
		    image: redis:5.0
		    ports:
		      - "6379:6379"
		    volumes:
		      - "/home/docker-redis/data:/data"
		      - "/home/docker-redis/conf:/usr/local/etc/redis"
		    deploy:
		      replicas: 1
		    command: redis-server /usr/local/etc/redis/redis.conf

		  zookeeper:
		    image: zookeeper:3.4
		    restart: always
		    ports:
		      - 2181:2181

	2.开始部署	      
		docker stack deploy -c base-stack.yml redis
		查看：
			docker stack ls
			docker service ls
			docker service ps redis-stack_redis
			docker service logs [服务名]
